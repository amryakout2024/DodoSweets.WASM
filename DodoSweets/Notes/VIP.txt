 @inject ProtectedLocalStorage BrowserStorage

async Task SaveTheme(string themeName) {
    await BrowserStorage.SetAsync("userTheme", themeName);
}

async Task<string> LoadTheme() {
    var result = await BrowserStorage.GetAsync<string>("userTheme");
    return result.Success ? result.Value : "light";
}
